## <a name="chapter-three" id="chapter-three"></a>三 解题思路

> [返回目录](#chapter-one)

* 递归

第一眼就是回溯：

```js
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
const combine = (n, k) => {
  // 1. 设置 result 获取结果值
  const result = [];
  
  // 2. 设置递归
  const recursion = (path) => {
    // 2.1 设置递归终止条件，即当数组长度为 k 的时候
    if (path.length === k) {
      result.push(path.concat());
      return;
    }

    // 2.2 获取最后一个数字，因为不确定是否有负整数，所以这里设置为最大负数
    const lastNumber = path[path.length - 1] || Number.MIN_SAFE_INTEGER;

    // 2.3 遍历 [1, n]，回溯添加数字
    for (let i = 1; i <= n; i++) {
      // 2.4 前提是当前组合不包含，以及比最后一个数字大
      if (!path.includes(i) && lastNumber < i) {
        path.push(i);
        recursion(path);
        path.pop();
      }
    }
  };
  recursion([]);

  // 3. 返回最终结果
  return result;
};

console.log(combine(4, 3)); // [ [ 1, 2, 3 ], [ 1, 2, 4 ], [ 1, 3, 4 ], [ 2, 3, 4 ] ]
```

* 递归（剪枝）

```js
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
const combine = (n, k) => {
  // 1. 设置 result 获取结果值
  const result = [];
  
  // 2. 设置递归
  const recursion = (current, path) => {
    // 2.1 剪枝：如果长度不可能凑够
    if (path.length + (n - current + 1) < k) {
      return;
    }
    // 2.2 终止：如果长度够了
    if (path.length === k) {
      result.push(path);
      return;
    }
    // 2.3 考虑选择当前位置
    recursion(current + 1, [...path, current]);
    // 2.4 考虑不选择当前位置
    recursion(current + 1, path);
  };
  recursion(1, []);

  // 3. 返回最终结果
  return result;
};

console.log(combine(4, 3)); // [ [ 1, 2, 3 ], [ 1, 2, 4 ], [ 1, 3, 4 ], [ 2, 3, 4 ] ]
```
